207 Stress Dislocation::stressFieldLocal (Vector3d p, double mu, double nu) const
208 {
209  double D = ( mu * this->bmag ) / ( 2.0 * PI * ( 1.0 - nu ) ); // Constant for all components of the stress tensor
210 
211  double x, y, denominator; // Terms that appear repeatedly in the stress tensor
212 
213  x = p.getValue (0);
214  y = p.getValue (1);
215  denominator = pow ( ((x*x) + (y*y)), 2);
216 
217  double principalStresses[3], shearStresses[3];
218 
219  principalStresses[0] = -1.0 * D * y * ( (3.0*x*x) + (y*y) ) / denominator;
220  principalStresses[1] = D * y * ( (x*x) - (y*y) ) / denominator;
221  principalStresses[2] = nu * ( principalStresses[0] + principalStresses[1] );
222 
223  shearStresses[0] = D * x * ( (x*x) - (y*y) ) / denominator;
224  shearStresses[1] = 0.0;
225  shearStresses[2] = 0.0;
226 
227  return (Stress(principalStresses, shearStresses));
228 }

262 double Dislocation::idealTimeIncrement (Vector3d v0, double minDistance, Defect d, Vector3d v1)
263 {

265  if (norm_v0 == 0.0)
266  {
267  // This dislocation is not moving
268  return (1000.0);
269  }
270 
271  // Positions

273  Vector3d p1 = d.getPosition();
274  Vector3d p01 = p1 - p0;
275  double norm_p01 = p01.magnitude();
276 
277  if (norm_p01 == 0.0)

279  // The dislocation is lying on top of the obstacle - so it should not move
280  return (0.0);
281  }
282  else
283  {
284  // Find out if the dislocation is approaching the defect or not
285 
286  // Velocities
287  Vector3d v01 = v1 - v0;

289  double dotProduct = v01 * p01;
290  double cosine = dotProduct/(norm_v01 * norm_p01);
