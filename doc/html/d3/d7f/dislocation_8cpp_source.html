294 Stress Dislocation::stressFieldLocal (Vector3d p, double mu, double nu) const
295 {
296  double D = ( mu * this->bmag ) / ( 2.0 * PI * ( 1.0 - nu ) ); // Constant for all components of the stress tensor
297 
298  double x, y, denominator; // Terms that appear repeatedly in the stress tensor
299 
300  x = p.getValue (0);
301  y = p.getValue (1);
302  denominator = pow ( ((x*x) + (y*y)), 2);
303 
304  double principalStresses[3], shearStresses[3];
305 
306  principalStresses[0] = -1.0 * D * y * ( (3.0*x*x) + (y*y) ) / denominator;
307  principalStresses[1] = D * y * ( (x*x) - (y*y) ) / denominator;
308  principalStresses[2] = nu * ( principalStresses[0] + principalStresses[1] );
309 
310  shearStresses[0] = D * x * ( (x*x) - (y*y) ) / denominator;
311  shearStresses[1] = 0.0;
312  shearStresses[2] = 0.0;
313 
314  return (Stress(principalStresses, shearStresses));
315 }

348 double Dislocation::idealTimeIncrement (double minDistance, Defect d, Vector3d v1)
349 {
350  Vector3d v0 = this->velocity;
351  double norm_v0 = v0.magnitude();
352  if (norm_v0 == 0.0)
353  {
354  // This dislocation is not moving
355  return (1000.0);
356  }
357 
358  // Positions
359  Vector3d p0 = this->getPosition();
360  Vector3d p1 = d.getPosition();
361  Vector3d p01 = p1 - p0;
362  double norm_p01 = p01.magnitude();
363 
364  if (norm_p01 == 0.0)
365  {
366  // The dislocation is lying on top of the obstacle - so it should not move
367  return (0.0);
368  }
369  else
370  {
371  // Find out if the dislocation is approaching the defect or not
372 
373  // Velocities
374  Vector3d v01 = v1 - v0;
375  double norm_v01 = v01.magnitude();
376  double dotProduct = v01 * p01;
377  double cosine = dotProduct/(norm_v01 * norm_p01);
378  if (cosine < 0.0)
379  {
380  // The dislocation is approaching the other defect
381  return ( (norm_p01 - minDistance)/norm_v01 );
382  }
383  else
384  {
385  // They are diverging
386  // So any time increment will do
387  return (1000.0);
388  }
389  }
