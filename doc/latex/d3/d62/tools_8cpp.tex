  std::vector<Vector3d>::iterator vi;
  std::vector<Vector3d>::iterator vj;

  bool found;
  int i;

  std::vector<Vector3d> vList;
  Vector3d vMinus;

  if (negatives)
    {
      Vector3d vPlus;
    }

  for (vi=v.begin();vi!=v.end();vi++)
    {
      for (vj=vi+1; vj!=v.end(); vj++)
        {
          vMinus = (*vi) - (*vj);

          found = (vMinus.magnitude()==0.0);
          if (negatives)
            {
              vPlus  = (*vi) + (*vj);
              found = found || (vPlus.magnitude()==0.0);
            }

          if (found)
            {
              break;
            }
        }
      if (!found)
        {
          vList.push_back(*vi);
        }
    }

  return (vList);
}
